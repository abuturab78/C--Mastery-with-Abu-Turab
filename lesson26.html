Lesson Details:

Is lesson mein hum ek simple student record system banayenge jismein user students ke records add kar sakta hai, display kar sakta hai, aur search kar sakta hai.

Key Points:

  • Student Structure: Har student ka record store karne ke liye ek structure banayenge. 
                                
struct Student {
    int rollNo;
    string name;
    float marks;
};
  • Add Record: User naye students ke records add kar sakta hai. 
                                
Student s;
cout << "Enter roll number: ";
cin >> s.rollNo;
cout << "Enter name: ";
cin >> s.name;
cout << "Enter marks: ";
cin >> s.marks;
students.push_back(s);
  • Display Records: User sabhi students ke records display kar sakta hai. 
                                
for (const auto& s : students) {
    cout << "Roll No: " << s.rollNo << ", Name: " << s.name << ", Marks: " << s.marks << endl;
}
  • Search Record: User kisi specific student ka record search kar sakta hai. 
                                
int rollNo;
cout << "Enter roll number to search: ";
cin >> rollNo;
for (const auto& s : students) {
    if (s.rollNo == rollNo) {
        cout << "Record found: " << s.name << ", Marks: " << s.marks << endl;
        break;
    }
}
  • Example: Ek complete program jo student record system ko implement karta hai: 
                                
#include <iostream>
#include <vector>
#include <string>
using namespace std;

struct Student {
    int rollNo;
    string name;
    float marks;
};

int main() {
    vector<Student> students;
    int choice;

    while (true) {
        cout << "\n1. Add Record\n2. Display Records\n3. Search Record\n4. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1: {
                Student s;
                cout << "Enter roll number: ";
                cin >> s.rollNo;
                cout << "Enter name: ";
                cin >> s.name;
                cout << "Enter marks: ";
                cin >> s.marks;
                students.push_back(s);
                break;
            }
            case 2: {
                for (const auto& s : students) {
                    cout << "Roll No: " << s.rollNo << ", Name: " << s.name << ", Marks: " << s.marks << endl;
                }
                break;
            }
            case 3: {
                int rollNo;
                cout << "Enter roll number to search: ";
                cin >> rollNo;
                for (const auto& s : students) {
                    if (s.rollNo == rollNo) {
                        cout << "Record found: " << s.name << ", Marks: " << s.marks << endl;
                        break;
                    }
                }
                break;
            }
            case 4: {
                cout << "Thank you for using our student record system!";
                return 0;
            }
            default: {
                cout << "Invalid choice!";
            }
        }
    }

    return 0;
}
