Lesson Details:

Tic-Tac-Toe ek classic game hai jismein do players ek 3x3 grid mein 'X' aur 'O' daalte hain. Is lesson mein hum yeh game C++ mein banayenge.

Key Points:

  • Game Board: Game board ko 3x3 grid ke roop mein represent kiya jata hai. 
                                
char board[3][3] = {
    {'1', '2', '3'},
    {'4', '5', '6'},
    {'7', '8', '9'}
};
  • Player Moves: Players apne moves grid ke numbers ke through select karte hain. 
                                
int move;
cout << "Enter your move (1-9): ";
cin >> move;
  • Win Condition: Game ka winner check karne ke liye rows, columns, aur diagonals ko check kiya jata hai. 
                                
bool checkWin() {
    // Check rows, columns, and diagonals
    for (int i = 0; i < 3; i++) {
        if (board[i][0] == board[i][1] && board[i][1] == board[i][2]) return true;
        if (board[0][i] == board[1][i] && board[1][i] == board[2][i]) return true;
    }
    if (board[0][0] == board[1][1] && board[1][1] == board[2][2]) return true;
    if (board[0][2] == board[1][1] && board[1][1] == board[2][0]) return true;
    return false;
}
  • Example: Ek complete Tic-Tac-Toe game ka code: 
                                
#include <iostream>

using namespace std;

char board[3][3] = {
    {'1', '2', '3'},
    {'4', '5', '6'},
    {'7', '8', '9'}
};

void displayBoard() {
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            cout << board[i][j] << " ";
        }
        cout << endl;
    }
}

bool checkWin() {
    // Check rows, columns, and diagonals
    for (int i = 0; i < 3; i++) {
        if (board[i][0] == board[i][1] && board[i][1] == board[i][2]) return true;
        if (board[0][i] == board[1][i] && board[1][i] == board[2][i]) return true;
    }
    if (board[0][0] == board[1][1] && board[1][1] == board[2][2]) return true;
    if (board[0][2] == board[1][1] && board[1][1] == board[2][0]) return true;
    return false;
}

int main() {
    int move;
    char player = 'X';

    cout << "Welcome to Tic-Tac-Toe!" << endl;

    for (int turn = 0; turn < 9; turn++) {
        displayBoard();
        cout << "Player " << player << ", enter your move (1-9): ";
        cin >> move;

        int row = (move - 1) / 3;
        int col = (move - 1) % 3;

        if (board[row][col] != 'X' && board[row][col] != 'O') {
            board[row][col] = player;
            if (checkWin()) {
                displayBoard();
                cout << "Player " << player << " wins!" << endl;
                return 0;
            }
            player = (player == 'X') ? 'O' : 'X';
        } else {
            cout << "Invalid move! Try again." << endl;
            turn--;
        }
    }

    displayBoard();
    cout << "It's a draw!" << endl;

    return 0;
}
