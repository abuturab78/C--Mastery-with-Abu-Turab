#include <iostream>
#include <vector>
#include <string>
using namespace std;

struct Student {
    int id;
    string name;
    float marks;
};

void addStudent(vector<Student>& students) {
    Student s;
    cout << "Enter student ID: ";
    cin >> s.id;
    cout << "Enter student name: ";
    cin >> s.name;
    cout << "Enter student marks: ";
    cin >> s.marks;
    students.push_back(s);
}

void displayStudents(const vector<Student>& students) {
    for (const auto& s : students) {
        cout << "ID: " << s.id << ", Name: " << s.name << ", Marks: " << s.marks << endl;
    }
}

int main() {
    vector<Student> students;
    int choice;

    while (true) {
        cout << "\n1. Add Student\n2. Display Students\n3. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        if (choice == 1) {
            addStudent(students);
        } else if (choice == 2) {
            displayStudents(students);
        } else if (choice == 3) {
            break;
        } else {
            cout << "Invalid choice!";
        }
    }

    return 0;
}
  • Recap: C++ programming ke sabhi important concepts ka recap karenge.
    • Variables and Data Types
    • Input and Output
    • Conditional Statements
    • Loops
    • Functions
    • Arrays and Vectors
    • Structures
    • File Handling
